# Queen Elizabeth I (1533-1603)
country_event = {
	id = women_history_events.1
	title = women_history_events.1.t
	desc = women_history_events.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	
	trigger = {
		is_year = 1533
		NOT = {
			is_year = 1603
		}
		OR = {
			tag = ENG
			tag = GBR
		}
		is_female = no
		has_regency = no
		has_heir = no
		is_lesser_in_union = no
		government = monarchy
		religion = catholic
		any_owned_province = {
			religion = protestant
			is_in_capital_area = yes
		}
		NOT = {
			has_global_flag = wih_elizabeth_eng_flag
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = wih_elizabeth_eng_flag
		}
	}
	
	option = {
		name = women_history_events.1.a
		change_religion = protestant
		define_ruler = {
			name = "Elizabeth"
			dynasty = "Tudor"
			adm = 6
			dip = 6
			mil = 5
			female = yes
		}
	}
	option = {
		name = women_history_events.1.b
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		define_advisor = {
			female = yes
			type = statesman #spymaster
			skill = 2
			name = "Elizabeth Tudor"
		}
	}
}
