country_event = {
	id = women_history_events.3
	title = women_history_events.3.t
	desc = women_history_events.3.d
	picture = wihgfx_FEMALE_RULER_eventPicture

	trigger = {
		tag = TUR
		is_year = 1590

	}

		hidden_effect = {

		}

		}
	}
