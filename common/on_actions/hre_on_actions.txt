# country
on_startup = {
	if = {
		limit = {
			hre_size = 1
			is_emperor = yes
		}
		country_event = {
		    id = hre_startup.0
		}
	}
	if = {
		limit = {
			is_year = 1495
			NOT = { has_global_flag = had_diet1495 }
			NOT = { is_year = 1496 }
		}
		country_event = {
			id = hre_reforms.0
			days = 30
			random = 10
		}
	}
	if = {
		limit = {
			has_country_flag = spokesman_diet
		}
		country_event = {
			id = hre_reforms.7
			days = 90
			random = 10
		}
	}
	if = {
		limit = {
			is_part_of_hre = yes
			is_emperor = no
			OR = {
				AND = {
					is_year = 1505
					NOT = { is_year = 1506 }
				}
				AND = {
					is_year = 1512
					NOT = { is_year = 1513 }
				}
			}
		}
		country_event = {
			id = hre_reforms.13
			days = 30
			random = 10
		}
	}
}
